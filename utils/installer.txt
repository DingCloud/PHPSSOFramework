CREATE TABLE  `ssodb`.`users` (
`uid` INT( 11 ) NOT NULL AUTO_INCREMENT COMMENT  'user_id',
`username` VARCHAR( 100 ) NOT NULL ,
`password` VARCHAR( 100 ) NOT NULL ,
`usertoken` VARCHAR( 100 ) NOT NULL ,
`ureg_time` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP ,
PRIMARY KEY (  `uid` )
) ENGINE = MYISAM ;
